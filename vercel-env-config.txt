# Vercel环境变量配置文件
# 请将以下环境变量复制到Vercel项目设置中的Environment Variables部分

# 飞书机器人配置
FEISHU_WEBHOOK_URL=https://open.larksuite.com/open-apis/bot/v2/hook/3900791f-0dfd-4821-8d5f-58d158487db7
FEISHU_SECRET=

# Next.js 配置 - 线上网址
NEXT_PUBLIC_SITE_URL=https://www.sportstreamhd.com

# 配置说明：
# 1. FEISHU_WEBHOOK_URL: 飞书机器人的Webhook地址，用于接收表单提交通知
# 2. FEISHU_SECRET: 飞书机器人的签名密钥（当前为空，如需签名验证请填写）
# 3. NEXT_PUBLIC_SITE_URL: 网站的公开访问地址，用于生成正确的链接和元数据

# Vercel配置步骤：
# 1. 登录Vercel控制台
# 2. 进入你的项目设置页面
# 3. 点击"Environment Variables"选项卡
# 4. 逐个添加上述环境变量：
#    - Name: FEISHU_WEBHOOK_URL
#    - Value: https://open.larksuite.com/open-apis/bot/v2/hook/3900791f-0dfd-4821-8d5f-58d158487db7
#    - Environment: Production, Preview, Development (全选)
#    
#    - Name: FEISHU_SECRET
#    - Value: (留空或填写你的密钥)
#    - Environment: Production, Preview, Development (全选)
#    
#    - Name: NEXT_PUBLIC_SITE_URL
#    - Value: https://www.sportstreamhd.com
#    - Environment: Production, Preview, Development (全选)
# 5. 保存后重新部署项目

# 注意事项：
# - NEXT_PUBLIC_ 前缀的变量会暴露给客户端，请确保不包含敏感信息
# - 配置完成后需要重新部署才能生效
# - 建议在Preview和Development环境也配置相同变量以确保一致性
